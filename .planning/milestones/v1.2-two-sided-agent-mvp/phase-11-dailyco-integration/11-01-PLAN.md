```
// Skills referenced:
//   cloudflare/skills: durable-objects — per-employee room ownership
//   cloudflare/skills: cloudflare — Workers fetch() to Daily.co REST API

const DAILY_BASE = "https://api.daily.co/v1";

export interface DailyRoom {
  id: string;
  name: string;
  url: string;
  privacy: "public" | "private";
  created_at: string;
}

export interface DailyMeetingToken {
  token: string;
}

// createRoom: creates a Daily.co room with the given name.
//   - privacy: "private" so only token-holders can join.
//   - properties.exp: if provided (Unix seconds), the room auto-expires.
//   - Returns null (no throw) when DAILY_API_KEY is absent.
export async function createRoom(
  apiKey: string | undefined,
  name: string,
  options?: { exp?: number }
): Promise<DailyRoom | null>

// getRoom: fetches an existing room by name. Returns null when absent or key missing.
export async function getRoom(
  apiKey: string | undefined,
  name: string
): Promise<DailyRoom | null>

// deleteRoom: deletes a room. Returns { deleted: true } or null on failure.
export async function deleteRoom(
  apiKey: string | undefined,
  name: string
): Promise<{ deleted: true } | null>

// getMeetingToken: mints a per-user meeting token for roomName.
//   - is_owner: true for the room owner, false for guests.
//   - exp: optional Unix seconds expiry for the token.
export async function getMeetingToken(
  apiKey: string | undefined,
  roomName: string,
  options: { is_owner: boolean; user_name?: string; exp?: number }
): Promise<DailyMeetingToken | null>
```

All four functions share a private `dailyFetch(apiKey, path, opts)` helper that:
- Returns null immediately when apiKey is undefined/empty (fail-soft).
- Sends `Authorization: Bearer {apiKey}` header.
- Returns null (not throws) on non-2xx responses, logging the status via `console.error("daily.co", status, text)`.

Add the `ensurePersonalRoom(apiKey: string | undefined)` async method to `EmployeeMailboxDO`. Place it after `sendPushToSubscriptions` and before `emailToChat`. It:

1. Reads `personal_room_url` from the profile row: `SELECT personal_room_url FROM profile WHERE id = 1`
2. If non-null, returns `{ ok: true, url: existing_url, name: existing_name }` immediately (idempotent).
3. Reads `employee_id` and `display_name` from the profile row.
4. Derives the room name slug: `"parrot-" + employee_id` (Clerk user IDs are already URL-safe). Keep it short and deterministic.
5. Calls `createRoom(apiKey, roomName)` (privacy: "private", no exp = always-on personal room).
6. If createRoom returns null (key absent or Daily.co error): returns `{ ok: false, error: "room_provisioning_unavailable" }`.
7. On success: UPDATE profile SET `personal_room_name = roomName, personal_room_url = room.url` WHERE id = 1.
8. Returns `{ ok: true, url: room.url, name: roomName }`.

```typescript
// POST /api/meetings/ensure-room (Phase 11 Wave 1)
// Lazily provisions the signed-in employee's personal Daily.co room on first
// call; subsequent calls return the stored URL from the DO (no extra Daily.co
// call). Falls back gracefully when DAILY_API_KEY is absent.
app.post(
  "/api/meetings/ensure-room",
  requireEmployeeMailbox,
  async (c: AppContext) => {
    const result = await c.var.mailboxStub.ensurePersonalRoom(c.env.DAILY_API_KEY);
    if (!result.ok) {
      // Key absent or Daily.co error — degrade to Phase 13 toast path.
      return c.json({ ok: false, error: result.error }, 503);
    }
    return c.json({ ok: true, url: result.url, name: result.name });
  },
);
```

```typescript
// ── Phase 11 Daily.co smoke test (dev-only) ─────────────────────────
// Hit with:
//   curl -X POST http://localhost:8787/api/dev/smoke/dailyco \
//     -H "x-employee-id: dev-employee-1"
//
// pass: true  = DAILY_API_KEY present + createRoom + getRoom + deleteRoom
//               all succeed against the real Daily.co API.
// pass: false = key absent OR any call failed — logs reason, no throw.
app.post(
  "/api/dev/smoke/dailyco",
  async (c) => {
    if (!c.env.PARROT_DEV_MODE) {
      return c.json({ error: "dev mode only" }, 403);
    }
    const apiKey = c.env.DAILY_API_KEY;
    if (!apiKey) {
      return c.json({
        pass: false,
        reason: "DAILY_API_KEY not set — set via wrangler secret put DAILY_API_KEY",
      });
    }
    const roomName = `parrot-smoke-${Date.now()}`;
    const created = await createRoom(apiKey, roomName);
    if (!created) return c.json({ pass: false, reason: "createRoom failed" });
    const fetched = await getRoom(apiKey, roomName);
    const deleted = await deleteRoom(apiKey, roomName);
    return c.json({
      pass: !!fetched && !!deleted,
      room_name: roomName,
      room_url: created.url,
      get_ok: !!fetched,
      delete_ok: !!deleted,
    });
  },
);
```
