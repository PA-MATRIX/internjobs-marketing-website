**C. Add PATCH "/:id/flags" handler:**
```ts
adminEmployees.patch("/:id/flags", async (c) => {
  const id = c.req.param("id");
  const body = await c.req.json().catch(() => null);
  const FlagsSchema = z.object({
    featureFlags: z.object({
      email: z.boolean(),
      chat: z.boolean(),
      meetings: z.boolean(),
      phone: z.boolean(),
      sms: z.boolean(),
      campaigns: z.boolean(),
    }).partial(),
  });
  const parsed = FlagsSchema.safeParse(body);
  if (!parsed.success) {
    return c.json({ error: "invalid_body", details: parsed.error.flatten() }, 400);
  }
  const workspace = getWorkspaceStub(c.env);
  const row = await workspace.getEmployeeById(id);
  if (!row) return c.json({ error: "not_found" }, 404);

  // Merge with existing flags from KV (read-modify-write).
  let existing: Record<string, boolean> = {
    email: true, chat: true, meetings: true, phone: true, sms: true, campaigns: true,
  };
  if (c.env.PARROT_FEATURE_FLAGS) {
    const stored = await c.env.PARROT_FEATURE_FLAGS.get(
      `employee:${row.clerk_user_id}:flags`,
      { type: "json" },
    ) as Record<string, boolean> | null;
    if (stored) existing = { ...existing, ...stored };
  }
  const merged = { ...existing, ...parsed.data.featureFlags };
  if (c.env.PARROT_FEATURE_FLAGS) {
    await c.env.PARROT_FEATURE_FLAGS.put(
      `employee:${row.clerk_user_id}:flags`,
      JSON.stringify(merged),
    );
  }
  return c.json({ ok: true, employee_id: id, feature_flags: merged });
});
```

Also add GET "/:id/flags" for the frontend to read current flags:
```ts
adminEmployees.get("/:id/flags", async (c) => {
  const id = c.req.param("id");
  const workspace = getWorkspaceStub(c.env);
  const row = await workspace.getEmployeeById(id);
  if (!row) return c.json({ error: "not_found" }, 404);
  const defaults = { email: true, chat: true, meetings: true, phone: true, sms: true, campaigns: true };
  if (!c.env.PARROT_FEATURE_FLAGS) return c.json({ feature_flags: defaults });
  const stored = await c.env.PARROT_FEATURE_FLAGS.get(
    `employee:${row.clerk_user_id}:flags`,
    { type: "json" },
  ) as Record<string, boolean> | null;
  return c.json({ feature_flags: { ...defaults, ...(stored ?? {}) } });
});
```

Note: route order matters in Hono — register `"/:id/flags"` handlers BEFORE `"/:id"` (DELETE) to avoid the colon segment swallowing "/flags".
