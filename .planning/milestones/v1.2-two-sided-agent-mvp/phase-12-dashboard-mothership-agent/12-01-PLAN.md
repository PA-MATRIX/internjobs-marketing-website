```typescript
// v1.2 Phase 12 Wave 1: Workers AI direct REST helper for todo extraction.
//
// Pattern mirrors apps/app/src/workflows/student-inbound.mjs direct REST
// pattern (account 0fffd3dc637bdb26d4963df445a69fd3).
// Model: @cf/moonshotai/kimi-k2.6 — 262k context, response_format JSON schema,
// $0.95/$4.00 per million tokens (~$0.001/extraction at v1.2 email volumes).
//
// Fallback: if kimi-k2.6 errors (non-200 or parse fail), caller catches and
// logs; no todo is written but the source email/chat message is NOT blocked.
// The source message remains readable in its pane.
//
// Skills referenced:
//   cloudflare/skills: cloudflare — Workers AI direct REST, account binding pattern
//   cloudflare/skills: agents-sdk — Agent state, scheduling, onMessage patterns
//   cloudflare/skills: durable-objects — DO alarm self-reschedule, idempotent inserts

import type { Env } from "../types";

export interface ExtractedTodo {
  title: string;
  preview?: string;
  urgency_score: number;   // 0-100 LLM-assigned
  deadline_at?: string | null;
  mentioned_actors?: string[];
  is_mention: boolean;
}

export const TODO_EXTRACTION_SCHEMA = {
  type: "object",
  properties: {
    todos: {
      type: "array",
      items: {
        type: "object",
        required: ["title", "urgency_score", "is_mention"],
        properties: {
          title: { type: "string", maxLength: 120 },
          preview: { type: "string", maxLength: 300 },
          urgency_score: { type: "integer", minimum: 0, maximum: 100 },
          deadline_at: { type: ["string", "null"] },
          mentioned_actors: { type: "array", items: { type: "string" } },
          is_mention: { type: "boolean" },
        },
      },
    },
  },
} as const;

/**
 * Extract action items from an email body or chat message text.
 *
 * Returns [] on any error — caller must NOT let extraction failure block
 * email storage or alarm rescheduling.
 */
export async function extractTodosFromText(
  text: string,
  env: Env,
): Promise<ExtractedTodo[]> {
  try {
    const model = env.KIMI_MODEL ?? "@cf/moonshotai/kimi-k2.6";
    const accountId = env.CLOUDFLARE_ACCOUNT_ID;
    const apiToken = env.CLOUDFLARE_AI_API_TOKEN;

    if (!accountId || !apiToken) {
      console.warn("extractTodosFromText: CLOUDFLARE_AI_API_TOKEN or CLOUDFLARE_ACCOUNT_ID missing — skipping extraction");
      return [];
    }

    const resp = await fetch(
      `https://api.cloudflare.com/client/v4/accounts/${accountId}/ai/run/${encodeURIComponent(model)}`,
      {
        method: "POST",
        headers: {
          Authorization: `Bearer ${apiToken}`,
          "Content-Type": "application/json",
        },
        body: JSON.stringify({
          messages: [
            {
              role: "system",
              content:
                "You are an action-item extractor for a workplace assistant. " +
                "Given a message, extract concrete todos that require follow-up. " +
                "Assign urgency_score 0-100 based on language cues: 'urgent'/'ASAP'/'blocking' = 80-100, " +
                "'please reply'/'by EOD' = 60-79, neutral = 40-59, FYI/low-stakes = 0-39. " +
                "Set is_mention=true if the message @-mentions or directly addresses the recipient. " +
                "Return ONLY valid JSON matching the schema. Return {\"todos\":[]} if none.",
            },
            { role: "user", content: text.slice(0, 8000) },
          ],
          response_format: {
            type: "json_schema",
            json_schema: TODO_EXTRACTION_SCHEMA,
          },
          max_tokens: 512,
        }),
      },
    );

    if (!resp.ok) {
      console.error(`extractTodosFromText: AI REST ${resp.status} — falling back to no-todo`);
      return [];
    }

    const data = (await resp.json()) as { result?: { response?: string }; success?: boolean };
    if (!data?.result?.response) return [];

    const parsed = JSON.parse(data.result.response) as { todos?: ExtractedTodo[] };
    return Array.isArray(parsed?.todos) ? parsed.todos : [];
  } catch (err) {
    console.error("extractTodosFromText: unexpected error", err);
    return [];
  }
}
```
