Add a dev-only ranking regression endpoint to confirm that todo ordering respects the hybrid rank formula. This endpoint creates two todos directly in the DO (bypassing LLM extraction) and asserts the higher-urgency todo ranks first.

Add to `workers/index.ts` immediately after the existing smoke test endpoint:

```typescript
// ── Phase 12 ranking regression (dev-only) ──────────────────────
// Hit with: curl -X POST http://localhost:8787/api/dev/smoke/ranking \
//   -H "X-Parrot-Dev-Employee: dev@internjobs.ai"
//
// Asserts: urgency=80 todo ranks above urgency=20 is_mention=1 todo.
// Expected ranks:
//   todo-hi: (80*2) - recency_decay ≈ 160
//   todo-lo: (20*2) + 30 - recency_decay ≈ 70
// (Both created in same request so recency_decay is ~0 for both.)

app.post(
  "/api/dev/smoke/ranking",
  requireEmployeeMailbox,
  async (c: AppContext) => {
    if (!c.env.PARROT_DEV_MODE) {
      return c.json({ error: "dev-only endpoint" }, 403);
    }

    const stub = c.var.mailboxStub;
    const employee = c.var.employee;

    // Seed two todos with known scores via the createEmail path
    const hiId = `rank-hi-${Date.now()}`;
    const loId = `rank-lo-${Date.now() + 1}`;

    // Seed email that should produce urgency~80 todo
    await stub.createEmail(
      "Inbox",
      {
        id: hiId,
        subject: "URGENT: server is down, please fix immediately",
        sender: "ops@example.com",
        recipient: employee.email,
        date: new Date().toISOString(),
        body: "The production server is down. This is blocking all customers. Please fix ASAP — this is critical.",
      },
      [],
    );

    // Seed email that @-mentions the employee but is low urgency
    await stub.createEmail(
      "Inbox",
      {
        id: loId,
        subject: `FYI @${employee.displayName} — next team lunch is Thursday`,
        sender: "social@example.com",
        recipient: employee.email,
        date: new Date().toISOString(),
        body: `Hey @${employee.displayName}, just a heads up that the next team lunch is Thursday. No action needed!`,
      },
      [],
    );

    await new Promise((r) => setTimeout(r, 800));

    const todos = (await stub.getTodos("all")) as Array<{
      source_id: string;
      urgency_score: number;
      is_mention: boolean;
      rank: number;
    }>;

    const hiTodo = todos.find((t) => t.source_id === hiId);
    const loTodo = todos.find((t) => t.source_id === loId);

    const hiRanksFirst =
      hiTodo && loTodo ? hiTodo.rank > loTodo.rank : null;

    return c.json({
      hi_todo: hiTodo ?? null,
      lo_todo: loTodo ?? null,
      hi_ranks_first: hiRanksFirst,
      pass: hiRanksFirst === true,
    });
  },
);
```

Note: This endpoint relies on the LLM correctly assigning urgency_score=80+ to the "URGENT: server is down" email and a lower score to the FYI email. Since real LLM calls are used (not mocks), the exact scores will vary — the test asserts RELATIVE ranking, not absolute values. If both todos extracted have the same urgency_score, mention-boost may tip the balance. The `pass: true` assertion requires hiTodo.rank > loTodo.rank.

No new npm packages. No changes to non-index.ts files in this task.
