In `durableObject/index.ts`, add two new methods to `EmployeeMailboxDO`:

**Method `emailToChat(emailId: string): Promise<{ok: boolean; channel_url?: string; channel_id?: string; error?: string}>`**

Logic:
1. Fetch the email row by ID from the `emails` table: `SELECT id, subject, sender, recipient, body FROM emails WHERE id = ?`.
2. If not found, return `{ ok: false, error: 'email_not_found' }`.
3. Parse participants: collect sender + recipient as a deduplicated array of email addresses.
4. Build a channel name: slugify the email subject (lowercase, replace non-alphanumeric with `-`, truncate to 60 chars). If subject is null/empty use `email-${emailId.slice(0,8)}`.
5. Create a Mattermost channel via bot REST API:
   ```
   POST {env.MATTERMOST_URL}/api/v4/channels
   Authorization: Bearer {env.MATTERMOST_BOT_TOKEN}
   Body: { team_id: <first team the bot can see>, type: 'P' (private), display_name: <channel name>, name: <slugified> }
   ```
   Use `getMmChannelsForUser` pattern from `workers/lib/mattermost.ts` for the team lookup — or simply: first call `GET /api/v4/teams` with bot token and use `teams[0].id`. If team lookup fails or no MATTERMOST_BOT_TOKEN, return `{ ok: false, error: 'mattermost_unavailable' }`.
6. Post the email body as the seed message:
   ```
   POST {env.MATTERMOST_URL}/api/v4/posts
   Authorization: Bearer {env.MATTERMOST_BOT_TOKEN}
   Body: { channel_id: <new_channel_id>, message: "**Email from " + sender + "**\n\n" + body.slice(0,2000) }
   ```
7. Return `{ ok: true, channel_id: <id>, channel_url: env.MATTERMOST_URL + '/internjobs/channels/' + channel_name }`.
8. Wrap the entire Mattermost interaction in try/catch; on any error return `{ ok: false, error: 'mattermost_error' }`.
9. Note: `env` is accessible as `this.env` on the DO (inherited from `DurableObject<Env>` superclass).

**Method `chatToEmail(postId: string, postBody: string): Promise<{ok: boolean; draft?: {to: string; subject: string; body: string}; error?: string}>`**

Logic:
1. Get the employee profile via `getProfile()`.
2. Assemble the draft:
   ```typescript
   const draft = {
     to: '',          // empty — user fills in the To field in the compose modal
     subject: 'From chat: ' + postBody.slice(0, 60).replace(/\n/g, ' ') + (postBody.length > 60 ? '…' : ''),
     body: '> ' + postBody.split('\n').join('\n> ') + '\n\n',
   };
   ```
3. Return `{ ok: true, draft }`.

```tsx
// v1.2 Phase 13 Wave 2: EmailToChat — moves an email thread into a
// Mattermost channel seeded with the email body.
//
// Skills referenced:
//   cloudflare/skills: cloudflare — Workers AI via AI Gateway (per-employee quota + prompt cache)

import { useMutation } from "@tanstack/react-query";
import { useNavigate } from "react-router";
import { api } from "~/lib/api";

interface Props {
  emailId: string;
}

export function EmailToChat({ emailId }: Props) {
  const navigate = useNavigate();
  const action = useMutation({
    mutationFn: () => api.crosspaneEmailToChat(emailId),
    onSuccess: (data) => {
      // Store the channel URL so ChatPane can open it in the Mattermost iframe
      if (data.channel_url) {
        sessionStorage.setItem('parrot_crosspane_channel_url', data.channel_url);
      }
      navigate('/chat');
    },
  });

  return (
    <button
      type="button"
      onClick={() => action.mutate()}
      disabled={action.isPending}
      title="Move this email thread into a Mattermost channel"
      className="rounded-md border border-slate-300 bg-white px-3 py-1.5 text-xs font-medium text-slate-700 hover:bg-slate-50 disabled:opacity-50"
    >
      {action.isPending ? 'Moving…' : 'Move to Chat'}
    </button>
  );
}
```

```tsx
// v1.2 Phase 13 Wave 2: ChatToEmail — opens a draft compose modal
// pre-filled with the chat post body quoted.
//
// Skills referenced:
//   cloudflare/skills: cloudflare — Workers AI via AI Gateway (per-employee quota + prompt cache)

import { useState } from "react";
import { useMutation } from "@tanstack/react-query";
import { api } from "~/lib/api";

interface Props {
  postId?: string;
  postBody?: string;
}

interface DraftModal {
  to: string;
  subject: string;
  body: string;
}

export function ChatToEmail({ postId = '', postBody = '' }: Props) {
  const [draft, setDraft] = useState<DraftModal | null>(null);

  const action = useMutation({
    mutationFn: () => api.crosspaneChatToEmail(postId, postBody),
    onSuccess: (data) => {
      if (data.ok && data.draft) setDraft(data.draft);
    },
  });

  return (
    <>
      <button
        type="button"
        onClick={() => action.mutate()}
        disabled={action.isPending}
        title="Attach this chat post to an email draft"
        className="rounded-md border border-slate-300 bg-white px-3 py-1.5 text-xs font-medium text-slate-700 hover:bg-slate-50 disabled:opacity-50"
      >
        {action.isPending ? 'Preparing…' : 'Attach to Email'}
      </button>

      {draft && (
        <div className="fixed inset-0 z-50 flex items-center justify-center bg-black/40">
          <div className="w-full max-w-lg rounded-xl bg-white p-6 shadow-xl">
            <h2 className="text-sm font-semibold text-slate-900 mb-4">Compose Email</h2>
            <div className="space-y-3">
              <div>
                <label className="block text-xs text-slate-500 mb-1">To</label>
                <input
                  type="email"
                  value={draft.to}
                  onChange={(e) => setDraft({ ...draft, to: e.target.value })}
                  className="w-full rounded-md border border-slate-200 px-3 py-1.5 text-sm focus:outline-none focus:ring-1 focus:ring-slate-400"
                  placeholder="recipient@example.com"
                />
              </div>
              <div>
                <label className="block text-xs text-slate-500 mb-1">Subject</label>
                <input
                  type="text"
                  value={draft.subject}
                  onChange={(e) => setDraft({ ...draft, subject: e.target.value })}
                  className="w-full rounded-md border border-slate-200 px-3 py-1.5 text-sm focus:outline-none focus:ring-1 focus:ring-slate-400"
                />
              </div>
              <div>
                <label className="block text-xs text-slate-500 mb-1">Body</label>
                <textarea
                  rows={6}
                  value={draft.body}
                  onChange={(e) => setDraft({ ...draft, body: e.target.value })}
                  className="w-full rounded-md border border-slate-200 px-3 py-1.5 text-sm focus:outline-none focus:ring-1 focus:ring-slate-400"

              </div>
            </div>
            <div className="mt-4 flex gap-2 justify-end">
              <button
                type="button"
                onClick={() => setDraft(null)}
                className="rounded-md border border-slate-200 px-3 py-1.5 text-xs text-slate-600 hover:bg-slate-50"

                Cancel
              </button>
              <button
                type="button"
                onClick={() => {
                  // Navigate to /inbox with a ?compose=1 signal — the InboxPane
```
