    Calculate and verify WCAG contrast ratios for the three required brand pairs. Add contrast checks to the verify-brand.mjs script.

    WCAG contrast ratio formula:
    - Relative luminance: L = 0.2126 * R + 0.7152 * G + 0.0722 * B (each linearized)
    - Contrast ratio: (L1 + 0.05) / (L2 + 0.05) where L1 > L2

    Add a contrast checker to verify-brand.mjs (add before the Summary block):

    ```js
    // ─── BRAND-VERIFY-01: WCAG Contrast ──────────────────────────────────────────

    function hexToLinear(hex) {
      const v = parseInt(hex, 16) / 255;
      return v <= 0.04045 ? v / 12.92 : Math.pow((v + 0.055) / 1.055, 2.4);
    }

    function luminance(hex) {
      const r = hexToLinear(hex.slice(1, 3));
      const g = hexToLinear(hex.slice(3, 5));
      const b = hexToLinear(hex.slice(5, 7));
      return 0.2126 * r + 0.7152 * g + 0.0722 * b;
    }

    function contrastRatio(hex1, hex2) {
      const l1 = luminance(hex1);
      const l2 = luminance(hex2);
      const lighter = Math.max(l1, l2);
      const darker = Math.min(l1, l2);
      return (lighter + 0.05) / (darker + 0.05);
    }

    console.log("\n[BRAND-VERIFY-01] WCAG Contrast Ratios");
    const inkOnLavender = contrastRatio("#1A0D2E", "#E8DEF5");
    const inkOnLime = contrastRatio("#1A0D2E", "#CAFF4D");
    const lavenderOnCobalt = contrastRatio("#E8DEF5", "#3855FF");

    check(
      `Ink on lavender: ${inkOnLavender.toFixed(2)}:1 — WCAG AAA body (>= 7:1)`,
      inkOnLavender >= 7.0,
      `Ratio is ${inkOnLavender.toFixed(2)}:1 — below AAA threshold of 7:1`
    );
    check(
      `Ink on lime: ${inkOnLime.toFixed(2)}:1 — WCAG AA (>= 4.5:1)`,
      inkOnLime >= 4.5,
      `Ratio is ${inkOnLime.toFixed(2)}:1 — below AA threshold of 4.5:1`
    );
    check(
      `Lavender on cobalt: ${lavenderOnCobalt.toFixed(2)}:1 — WCAG AA (>= 4.5:1)`,
      lavenderOnCobalt >= 4.5,
      `Ratio is ${lavenderOnCobalt.toFixed(2)}:1 — below AA threshold of 4.5:1`
    );
    ```

    Run the updated script:

    If contrast check FAILs for ink-on-lavender (should not — these are the spec values, expected ratios are approximately 11:1 for ink/lavender): this indicates the color values in styles.css were entered incorrectly. Cross-check against BRAND-V1.md and fix the CSS variable values.

    Expected contrast ratios (for documentation):

    - Ink (#1A0D2E) on lime (#CAFF4D): approximately 8-10:1 (above AAA 7:1)
