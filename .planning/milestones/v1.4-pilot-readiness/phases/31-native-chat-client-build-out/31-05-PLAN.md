<objective>
Wave 4 — Real-time + notifications. Replaces the 5-second polling in ChatPane with a Worker-proxied WebSocket. The PAT never reaches the browser — the Worker sends the MM `authentication_challenge` server-side before bridging the connection. Adds presence/typing indicators, unread counts, in-app notification badges, and email notifications for @mentions/DMs when the tab is offline.

Purpose: Polling at 5s is noisy (6k requests/day per user) and gives terrible typing/presence UX. WebSocket delivery is the correct real-time primitive for a chat product. The PAT-never-in-browser constraint makes Worker-proxied WebSocket mandatory (research Decision 3).

IMPORTANT: Before Wave 4 runs in production, `chat.internjobs.ai` (the nginx proxy for MM) MUST pass WebSocket upgrade headers through to the Fly MM backend. Check this during Wave 4 prep. If the nginx proxy at chat.internjobs.ai blocks `Upgrade: websocket`, the Worker WS URL should be changed to use `MATTERMOST_WS_URL` pointing directly at `wss://internjobs-mattermost.fly.dev`. This is the open question from research (Question 3 in 31-RESEARCH.md) — resolve it and set the URL accordingly.

**Create new file `workers/lib/mm-ws-proxy.ts`:**

Implements the Worker-side WebSocket proxy per research Pattern 3. The PAT must be sent in `authentication_challenge` BEFORE the browser connection can send any data. The PAT must NEVER reach the browser — not in response frames, not in the authentication_challenge (which is sent upstream only).

```typescript
// workers/lib/mm-ws-proxy.ts
//
// Worker-proxied WebSocket for Mattermost real-time.
//
// Security invariant: the employee PAT NEVER reaches the browser. The Worker:
//   1. Resolves the employee PAT from WorkspaceDO
//   2. Opens an upstream WebSocket to MM
//   3. Sends authentication_challenge with the PAT on upstream open (upstream only)
//   4. Only then begins bidirectional proxying of non-auth frames
//
// Cloudflare Workers support WebSocket proxying via WebSocketPair.
// Reference: https://developers.cloudflare.com/workers/runtime-apis/websockets/
//
// IMPORTANT: chat.internjobs.ai must pass Upgrade: websocket headers through
// to the Fly MM backend. If it does not, set MATTERMOST_WS_URL in wrangler.jsonc
// to point directly at wss://internjobs-mattermost.fly.dev instead.

import { getWorkspaceStub } from "../durableObject/workspace";
import type { Env } from "../types";
import type { Employee } from "../types";

export async function handleChatWebSocket(
  request: Request,
  env: Env,
  employee: Employee,
): Promise<Response> {
  const upgradeHeader = request.headers.get("upgrade");
  if (upgradeHeader?.toLowerCase() !== "websocket") {
    return Response.json({ error: "Expected WebSocket upgrade" }, { status: 426 });
  }

  // Resolve employee PAT from WorkspaceDO
  const stub = getWorkspaceStub(env);
  const tokenRow = await stub.getEmployeeToken(employee.employeeId);
  if (!tokenRow) {
    return Response.json({ error: "chat_not_provisioned" }, { status: 503 });
  }

  // Create client-facing WebSocket pair
  const { 0: client, 1: server } = new WebSocketPair();

  // Derive WS URL from MATTERMOST_URL (https:// → wss://)
  // If chat.internjobs.ai doesn't pass WS upgrades, override with MATTERMOST_WS_URL env var.
  const mmWsUrl = (
    (env as unknown as { MATTERMOST_WS_URL?: string }).MATTERMOST_WS_URL ||
    env.MATTERMOST_URL.replace(/^https/, "wss").replace(/^http/, "ws")
  ) + "/api/v4/websocket";

  // Open upstream connection to Mattermost WebSocket
  const upstream = new WebSocket(mmWsUrl);

  upstream.addEventListener("open", () => {
    // Authenticate upstream with the PAT — this frame goes to MM only, never to the browser.
    upstream.send(JSON.stringify({
      seq: 1,
      action: "authentication_challenge",
      data: { token: tokenRow.token },
    }));
  });

  // Bidirectional proxy — no filtering needed; the PAT was only sent upstream above.
  upstream.addEventListener("message", (evt) => {
    if (server.readyState === WebSocket.OPEN) {
      server.send(evt.data);
    }
  });
  server.addEventListener("message", (evt) => {
    if (upstream.readyState === WebSocket.OPEN) {
      upstream.send(evt.data);
    }
  });

  upstream.addEventListener("close", (evt) => {
    server.close(evt.code, evt.reason);
  });
  server.addEventListener("close", (evt) => {
    upstream.close(evt.code, evt.reason);
  });

  upstream.addEventListener("error", () => {
    server.close(1011, "upstream_error");
  });

  // server.accept() takes NO arguments in the CF Workers runtime.
  server.accept();

  return new Response(null, {
    status: 101,
    webSocket: client,
  });
}
```

```typescript
function useChatWebSocket(opts: {
  enabled: boolean;
  onPosted: (post: MmPost) => void;
  onTyping: (channelId: string, userId: string) => void;
  onStatusChange: (userId: string, status: string) => void;
  onChannelUnread: (channelId: string) => void;
}) {
  const wsRef = useRef<WebSocket | null>(null);
  const reconnectTimeoutRef = useRef<ReturnType<typeof setTimeout>>();

  function connect() {
    if (wsRef.current?.readyState === WebSocket.OPEN) return;

    const protocol = window.location.protocol === "https:" ? "wss:" : "ws:";
    const ws = new WebSocket(`${protocol}//${window.location.host}/api/chat/ws`);

    ws.onmessage = (evt) => {
      try {
        const msg = JSON.parse(evt.data as string) as {
          event?: string;
          data?: Record<string, unknown>;
        };
        if (msg.event === "posted") {
          const post = JSON.parse((msg.data?.post as string) ?? "null") as MmPost | null;
          if (post) opts.onPosted(post);
        } else if (msg.event === "typing") {
          const channelId = msg.data?.channel_id as string;
          const userId = msg.data?.user_id as string;
          if (channelId && userId) opts.onTyping(channelId, userId);
        } else if (msg.event === "status_change") {
          const userId = msg.data?.user_id as string;
          const status = msg.data?.status as string;
          if (userId && status) opts.onStatusChange(userId, status);
        } else if (msg.event === "channel_viewed") {
          // unread cleared
        }
      } catch { /* malformed message — ignore */ }
    };

    ws.onclose = () => {
      // Reconnect with exponential backoff (2s → 4s → 8s, capped at 30s)
      reconnectTimeoutRef.current = setTimeout(connect, Math.min(2000 * (2 ** reconnectAttempts), 30000));
      reconnectAttempts = Math.min(reconnectAttempts + 1, 4);
    };

    ws.onerror = () => { ws.close(); };
    wsRef.current = ws;
  }

  let reconnectAttempts = 0;

  useEffect(() => {
    if (!opts.enabled) return;
    connect();
    return () => {
      clearTimeout(reconnectTimeoutRef.current);
      wsRef.current?.close(1000, "component_unmount");
    };
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [opts.enabled]);

  // Send typing event to MM via WS
  const sendTyping = useCallback((channelId: string) => {
    if (wsRef.current?.readyState === WebSocket.OPEN) {
      wsRef.current.send(JSON.stringify({
        seq: Date.now(),
        action: "user_typing",
        data: { channel_id: channelId, parent_id: "" },
      }));
    }
  }, []);

  return { sendTyping };
}
```
