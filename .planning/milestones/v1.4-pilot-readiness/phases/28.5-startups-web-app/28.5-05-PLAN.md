    Implement POST /webhooks/clerk for work-email enforcement:

    ```typescript
    import { Webhook } from "svix";

    // Personal domain blocklist (from RESEARCH.md §2)
    const BLOCKED_DOMAINS = new Set([
      "gmail.com", "yahoo.com", "hotmail.com", "outlook.com",
      "icloud.com", "aol.com", "proton.me", "live.com", "msn.com",
      "yahoo.co.uk", "googlemail.com"
    ]);

    function isPersonalEmail(email: string): boolean {
      const domain = email.split("@")[1]?.toLowerCase() ?? "";
      return BLOCKED_DOMAINS.has(domain) || domain.startsWith("gmx.");
    }

    export async function handleClerkWebhook(req: Request, env: Env): Promise<Response> {
      // 1. Svix signature verification
      const body = await req.text();
      const wh = new Webhook(env.STARTUPS_CLERK_WEBHOOK_SECRET);
      let payload: unknown;
      try {
        payload = wh.verify(body, {
          "svix-id": req.headers.get("svix-id") ?? "",
          "svix-timestamp": req.headers.get("svix-timestamp") ?? "",
          "svix-signature": req.headers.get("svix-signature") ?? "",
        });
      } catch {
        return new Response("invalid signature", { status: 400 });
      }

      const event = payload as { type: string; data: { id: string; email_addresses: Array<{ email_address: string }> } };

      // 2. Only handle user.created
      if (event.type !== "user.created") {
        return new Response("ok", { status: 200 });
      }

      // 3. Extract primary email
      const primaryEmail = event.data.email_addresses[0]?.email_address ?? "";

      // 4. Check domain
      if (isPersonalEmail(primaryEmail)) {
        // Delete the user — personal email domain is not allowed
        await fetch(`https://api.clerk.com/v1/users/${event.data.id}`, {
          method: "DELETE",
          headers: { Authorization: `Bearer ${env.STARTUPS_CLERK_SECRET_KEY}` },
        });
        // Log for audit (fire-and-forget via no ctx here — sync is fine for a fast DELETE)
        console.log(JSON.stringify({
          event: "work_email_enforcement",
          action: "deleted",
          user_id: event.data.id,
          email_domain: primaryEmail.split("@")[1],
        }));
        return new Response("ok", { status: 200 });
      }

      return new Response("ok", { status: 200 });
    }
    ```

    OPEN RISK (from RESEARCH.md §5): There is a race between user.created firing and the DELETE call
    completing — the user may briefly have a valid session. Clerk's native blocklist (paid tier) avoids
    this entirely. For v1.4, this webhook approach is the implementation. Add a code comment:
    `// TODO v1.5: migrate to Clerk paid-tier blocklist to eliminate the race condition`

    OPEN RISK (from RESEARCH.md §5): The email_addresses array may not include the primary email for

    if email_addresses is empty, do NOT delete the user (could be a false negative; Clerk's system
    webhook may send user.updated with the email shortly after). Log the case for monitoring.

    const BASE_URL = process.env.TEST_BASE_URL ?? "https://startups.internjobs.ai";

    test.describe("founder happy path", () => {
