    ```typescript
    export function mintSlug(companyName: string): string {
      return companyName
        .toLowerCase()
        .replace(/[^a-z0-9]+/g, "-")   // non-alphanumeric → hyphen
        .replace(/^-+|-+$/g, "")        // trim leading/trailing hyphens
        .slice(0, 30);                  // max 30 chars
    }

    // reserveUniqueSlug: calls the Fly proxy to check uniqueness.
    // apiUrl: STARTUP_API_URL env var; apiSecret: STARTUP_API_SECRET
    // Returns the reserved slug (never includes @startups.internjobs.ai — just the slug portion)
    export async function reserveUniqueSlug(
      base: string,
      apiUrl: string,
      apiSecret: string
    ): Promise<string> {
      const MAX_ATTEMPTS = 10;
      let slug = base;
      for (let attempt = 0; attempt < MAX_ATTEMPTS; attempt++) {
        const candidate = attempt === 0 ? base : `${base}-${attempt}`;
        const email = `${candidate}@startups.internjobs.ai`;
        const res = await fetch(`${apiUrl}/v1/startups/check-slug?agent_email=${encodeURIComponent(email)}`, {
          headers: { "X-Api-Secret": apiSecret }
        });
        if (res.status === 404) return candidate; // slug is free
        slug = candidate;
      }
      throw new Error(`slug collision: could not reserve a slug for "${base}" after ${MAX_ATTEMPTS} attempts`);
    }
    ```

    ```typescript
    import PostalMime from "postal-mime";

    export async function handleInboundEmail(
      message: ForwardableEmailMessage,
      env: Env,
      ctx: ExecutionContext
    ): Promise<void> {
      // 1. Extract slug from recipient address
      const toAddress = message.to; // e.g. "acme@startups.internjobs.ai"
      const slug = toAddress.split("@")[0]?.toLowerCase();

      if (!slug) {
        message.setReject("invalid recipient address");
        return;
      }

      // 2. Resolve startup_id via Fly proxy — startup_channel_links lookup
      const channelRes = await fetch(
        `${env.STARTUP_API_URL}/v1/channels/resolve?email=${encodeURIComponent(toAddress)}`,
        { headers: { "X-Api-Secret": env.STARTUP_API_SECRET } }
      );

      if (!channelRes.ok) {
        message.setReject("startup not found");
        return;
      }

      const { startup_id, member_id } = await channelRes.json<{
        startup_id: string;
        member_id: string | null;
      }>();

      // 3. Parse MIME body using postal-mime (same package as apps/parrot)
      const rawEmail = await streamToArrayBuffer(message.raw, message.rawSize);
      const parser = new PostalMime();
      const parsed = await parser.parse(rawEmail);

      // 4. Extract In-Reply-To for thread stitching (best-effort — may be absent)
      const inReplyTo = message.headers.get("in-reply-to") ?? null;
      const messageId = message.headers.get("message-id") ?? null;

      // 5. Insert inbound_messages row
      await fetch(`${env.STARTUP_API_URL}/v1/messages/inbound`, {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          "X-Api-Secret": env.STARTUP_API_SECRET,
        },
        body: JSON.stringify({
          startup_id,
          member_id,
          channel: "email",
          from_address: message.from,
          to_address: toAddress,
          subject: parsed.subject ?? "",
          body_text: parsed.text ?? "",
          body_html: parsed.html ?? "",
          in_reply_to: inReplyTo,
          message_id: messageId,
        }),
      });
    }

    // Helper: same as parrot's streamToArrayBuffer
    async function streamToArrayBuffer(
      stream: ReadableStream,
      streamSize: number
    ): Promise<ArrayBuffer> {
      const result = new Uint8Array(streamSize);
      let bytesRead = 0;
      const reader = stream.getReader();
      while (true) {
        const { done, value } = await reader.read();
        if (done) break;
        result.set(value, bytesRead);
        bytesRead += value.byteLength;
      }
      return result.buffer;
    }
    ```
