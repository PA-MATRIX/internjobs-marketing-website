2. Add a `handleActioned` function that InboxPane passes to EmailPanel as `onActioned`:

```ts
async function handleActioned(action: "archived" | "deleted" | "moved-to-trash") {
  const previousFolder = folder; // the folder we were viewing
  const previousId = selectedId; // the message that was actioned

  // Clear selection immediately so EmailPanel unmounts
  setSelectedId(null);

  // Invalidate with prefix match (exact:false is the React Query default).
  // queryKey ["parrot","inbox"] cascades to ALL sub-keys:
  //   ["parrot","inbox", folder]          — folder message lists
  //   ["parrot","inbox","message", id]    — individual message caches
  // This ensures folder counts and the list both refetch without a full reload.
  queryClient.invalidateQueries({ queryKey: ["parrot", "inbox"] });

  if (action === "archived") {
    showToast("Archived — Undo", async () => {
      if (previousId) {
        await api.moveMessage(previousId, previousFolder);
        queryClient.invalidateQueries({ queryKey: ["parrot", "inbox"] });
      }
      setToast(null);
    });
  } else if (action === "moved-to-trash") {
    showToast("Moved to Trash — Undo", async () => {
      if (previousId) {
        await api.moveMessage(previousId, previousFolder);
        queryClient.invalidateQueries({ queryKey: ["parrot", "inbox"] });
      }
      setToast(null);
    });
  } else {
    // hard-deleted: no undo possible
    showToast("Deleted permanently");
  }
}
```
