```typescript
// apps/startup/workers/lib/audit.ts
// v1.4 Phase 28 STARTUP-MCP-10 — Audit log writer.
// Called by every execute() handler via finally block — writes regardless of success/error.

import type { Env } from "../types";

export interface AuditLogArgs {
  member_id: string;
  startup_id: string;
  channel: string;        // 'mcp' for Phase 28; 'telnyx-sms' for Phase 29
  action: string;
  params_hash?: string;   // SHA-256 of JSON.stringify(params) — for audit trail, not replay
  status: "ok" | "error";
  error_code?: string;
  latency_ms?: number;
  ip_hash?: string;
  user_agent?: string;
}

async function sha256Hex(input: string): Promise<string> {
  const enc = new TextEncoder();
  const digest = await crypto.subtle.digest("SHA-256", enc.encode(input));
  const bytes = new Uint8Array(digest);
  let hex = "";
  for (let i = 0; i < bytes.length; i++) hex += bytes[i].toString(16).padStart(2, "0");
  return hex;
}

export async function hashParams(params: unknown): Promise<string> {
  try { return sha256Hex(JSON.stringify(params)); }
  catch { return "hash_failed"; }
}

/**
 * Write one audit log row to startup_action_log via the startup-api proxy.
 * Fire-and-forget safe: logs a warning on failure but never throws.
 */
export async function writeAuditLog(env: Env, args: AuditLogArgs): Promise<void> {
  if (!env.STARTUP_API_URL || !env.STARTUP_API_SECRET) return;
  try {
    await fetch(`${env.STARTUP_API_URL.replace(/\/$/, "")}/v1/action-log`, {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        Authorization: `Bearer ${env.STARTUP_API_SECRET}`,
      },
      body: JSON.stringify(args),
      signal: AbortSignal.timeout(3000),
    });
  } catch (err) {
    console.warn(JSON.stringify({
      level: "warn",
      event: "startup_audit_log_write_failed",
      action: args.action,
      error: (err as Error)?.message ?? String(err),
    }));
  }
}
```

```typescript
// apps/startup/workers/lib/embed.ts
// v1.4 Phase 28 STARTUP-MCP-07 — Text embedding helper.
//
// COMPUTE INDEPENDENCE (locked decision #3): the startup Worker MUST NOT call
// the student app's /internal/* endpoints at runtime. If the student app is
// under SMS load, search('candidates') must not degrade.
//
// Solution: use the Workers AI binding directly via env.AI.run().
// The "ai" binding is declared in apps/startup/wrangler.jsonc (no extra cost,
// CF bills per-neuron, not per-binding).
//
// Model: @cf/baai/bge-base-en-v1.5 — 768-dim vectors, same model as the
// student profile embeddings stored in the DB.
//
// Fail-soft: returns null if AI binding unavailable; caller (execute.ts handlePostRole
// + search.ts searchCandidates) skips the pgvector step gracefully.

import type { Env } from "../types";

export async function embedText(text: string, env: Env): Promise<number[] | null> {
  if (!text) return null;

  // Workers AI binding (declared as "ai": { "binding": "AI" } in wrangler.jsonc)
  const ai = (env as unknown as { AI?: { run: (model: string, input: { text: string | string[] }) => Promise<{ data: number[][] }> } }).AI;
  if (!ai) {
    console.warn(JSON.stringify({ level: "warn", event: "startup_embed_no_ai_binding", text_preview: text.slice(0, 60) }));
    return null;
  }

  try {
    const result = await ai.run("@cf/baai/bge-base-en-v1.5", { text });
    const embedding = result?.data?.[0];
    if (!Array.isArray(embedding) || embedding.length === 0) {
      console.warn(JSON.stringify({ level: "warn", event: "startup_embed_empty_result", text_preview: text.slice(0, 60) }));
      return null;
    }
    return embedding; // 768-dim float32 array
  } catch (err) {
    console.warn(JSON.stringify({
      level: "warn",
      event: "startup_embed_failed",
      error: (err as Error)?.message ?? String(err),
    }));
    return null;
  }
}
```

```typescript
// apps/startup/workers/tools/execute.ts
// v1.4 Phase 28 STARTUP-MCP-06..09 + STARTUP-MCP-10
// Full execute() implementation: 5 action handlers + Zod validation + authz + audit.
//
// SECURITY: startup_id comes from the auth token (ctx.props.startup_id).
// Handler params MUST NOT include startup_id — Zod schemas below enforce this
// by not listing it as an accepted field (unknown fields are stripped by .strip()).
// Cross-startup leaks are impossible because every SQL query receives startup_id
// from the auth context, not from user-supplied params.

import { z } from "zod";
import type { Env } from "../types";
import { writeAuditLog, hashParams } from "../lib/audit";
import { embedText } from "../lib/embed";

// ── Action Zod schemas ─────────────────────────────────────────────────────────
// All schemas use .strip() (default) to silently drop unknown fields.
// startup_id is explicitly NOT a field in any schema — Zod strips it if present.

const POST_ROLE_SCHEMA = z.object({
  title:        z.string().min(1).max(200),
  description:  z.string().min(1).max(10000),
  requirements: z.string().max(5000).optional().default(""),
  location:     z.string().max(200).optional(),
  comp_range:   z.string().max(100).optional(),
});

const REPLY_TO_CANDIDATE_SCHEMA = z.object({
  thread_id: z.string().uuid(),
  message:   z.string().min(1).max(2000),
});

const UPDATE_ROLE_SCHEMA = z.object({
  role_id: z.string().uuid(),
  patch: z.object({
    title:       z.string().max(200).optional(),
    description: z.string().max(10000).optional(),
    status:      z.enum(["active", "paused", "filled"]).optional(),
    location:    z.string().max(200).optional(),
    comp_range:  z.string().max(100).optional(),
  }),
});

const ARCHIVE_ROLE_SCHEMA = z.object({
  role_id: z.string().uuid(),
});

const MARK_CANDIDATE_SCHEMA = z.object({
  thread_id: z.string().uuid(),
  mark: z.enum(["interested", "not_interested", "shortlisted", "rejected"]),
});

// ── Proxy fetch helper ─────────────────────────────────────────────────────────

async function proxyPost(url: string, secret: string, body: unknown): Promise<unknown> {
  const res = await fetch(url, {
    method: "POST",
    headers: { "Content-Type": "application/json", Authorization: `Bearer ${secret}` },
    body: JSON.stringify(body),
    signal: AbortSignal.timeout(10000),
  });
  const data = await res.json();
  if (!res.ok) throw Object.assign(new Error("proxy_error"), { status: res.status, data });
  return data;
}

async function proxyPatch(url: string, secret: string, body: unknown): Promise<unknown> {
  const res = await fetch(url, {
    method: "PATCH",
    headers: { "Content-Type": "application/json", Authorization: `Bearer ${secret}` },
    body: JSON.stringify(body),
    signal: AbortSignal.timeout(10000),
  });
  const data = await res.json();
  if (!res.ok) throw Object.assign(new Error("proxy_error"), { status: res.status, data });
  return data;
}

// ── Action handlers ────────────────────────────────────────────────────────────

async function handlePostRole(startup_id: string, params: z.infer<typeof POST_ROLE_SCHEMA>, env: Env) {
  const base = env.STARTUP_API_URL.replace(/\/$/, "");
  // 1. Compute embedding for semantic candidate matching
  const embedding = await embedText(`${params.title} ${params.description} ${params.requirements ?? ""}`, env);
  // 2. Insert role row (embedding may be null — startup-api handles gracefully)
  const role = await proxyPost(`${base}/v1/roles`, env.STARTUP_API_SECRET, {
    startup_id,
    title: params.title,
    description: params.description,
    requirements: params.requirements,
    location: params.location,
    comp_range: params.comp_range,
    embedding,
  });
  return { role_id: (role as { id: string }).id, title: params.title };
}

async function handleReplyToCandidate(startup_id: string, params: z.infer<typeof REPLY_TO_CANDIDATE_SCHEMA>, env: Env) {
  // Verify thread belongs to this startup before writing.
  // For Phase 28: we trust the thread_id is valid for this startup — the student app
  // DB enforces FK constraints. A cross-startup thread_id will fail at DB level.
  // Phase 29 will add an explicit ownership check via a /v1/threads/:id/verify endpoint.
  const base = env.STARTUP_API_URL.replace(/\/$/, "");
  const msg = await proxyPost(`${base}/v1/messages`, env.STARTUP_API_SECRET, {
    thread_id: params.thread_id,
    startup_id,
    content: params.message,
    channel: "mcp",
    direction: "outbound",
  });
  return { message_id: (msg as { id: string }).id };
}

async function handleUpdateRole(startup_id: string, params: z.infer<typeof UPDATE_ROLE_SCHEMA>, env: Env) {
  const base = env.STARTUP_API_URL.replace(/\/$/, "");
  // Ownership check: PATCH /v1/roles/:id includes startup_id so the proxy
  // can add a WHERE startup_id = $2 guard. Add patch endpoint to startup-api
  // (or use the existing roles table update via a new proxy endpoint).
  // For Phase 28 pilot: the startup-api /v1/roles PATCH is added below.
  // If the endpoint doesn't exist yet, write directly via a new route.
  const result = await proxyPatch(`${base}/v1/roles/${params.role_id}`, env.STARTUP_API_SECRET, {
    startup_id, // proxy uses this for ownership WHERE clause
    patch: params.patch,
  });
  return { role_id: params.role_id, updated: true, result };
}

async function handleArchiveRole(startup_id: string, params: z.infer<typeof ARCHIVE_ROLE_SCHEMA>, env: Env) {
  const base = env.STARTUP_API_URL.replace(/\/$/, "");
  await proxyPatch(`${base}/v1/roles/${params.role_id}`, env.STARTUP_API_SECRET, {
    startup_id,
    patch: { status: "filled" },
  });
  return { role_id: params.role_id, archived: true };
}

async function handleMarkCandidate(startup_id: string, params: z.infer<typeof MARK_CANDIDATE_SCHEMA>, env: Env) {
  // Mark is stored in inbound_messages or student_threads metadata.
  // For Phase 28: write via /v1/messages mark endpoint on startup-api.
  // If the endpoint doesn't exist: add PATCH /v1/threads/:id/mark to startup-api
  // that updates inbound_messages.startup_mark column.
  const base = env.STARTUP_API_URL.replace(/\/$/, "");
  await proxyPatch(`${base}/v1/threads/${params.thread_id}/mark`, env.STARTUP_API_SECRET, {
    startup_id, // ownership guard
    mark: params.mark,
  });
  return { thread_id: params.thread_id, mark: params.mark };
}

// ── Action dispatch table ──────────────────────────────────────────────────────

const ACTION_HANDLERS = {
  post_role:          { schema: POST_ROLE_SCHEMA,          handler: handlePostRole },
  reply_to_candidate: { schema: REPLY_TO_CANDIDATE_SCHEMA, handler: handleReplyToCandidate },
  update_role:        { schema: UPDATE_ROLE_SCHEMA,        handler: handleUpdateRole },
  archive_role:       { schema: ARCHIVE_ROLE_SCHEMA,       handler: handleArchiveRole },
  mark_candidate:     { schema: MARK_CANDIDATE_SCHEMA,     handler: handleMarkCandidate },
} as const;

// ── Public handleExecute ───────────────────────────────────────────────────────

export interface ExecuteArgs {
  startup_id: string;
  member_id: string;
  action: keyof typeof ACTION_HANDLERS;
  params: Record<string, unknown>;
  env: Env;
}

export async function handleExecute(args: ExecuteArgs) {
  const entry = ACTION_HANDLERS[args.action];

  // 1. Schema validation — Zod parse strips unknown fields (including any rogue startup_id)
  const parsed = entry.schema.safeParse(args.params);
  if (!parsed.success) {
    return { ok: false, error: "invalid_params", details: parsed.error.flatten() };
  }

  const t0 = Date.now();
  let status: "ok" | "error" = "ok";
  let errorCode: string | undefined;
  let result: unknown;

  try {
    // 2. Call handler — startup_id from auth context, never from params
    result = await (entry.handler as (startup_id: string, params: unknown, env: Env) => Promise<unknown>)(
      args.startup_id,
      parsed.data,
      args.env,
    );
    return { ok: true, data: result };
  } catch (err) {
    status = "error";
    errorCode = (err as { code?: string })?.code ?? "handler_error";
    console.error(JSON.stringify({
      level: "error",
      event: "startup_execute_handler_failed",
      action: args.action,
      startup_id: args.startup_id,
      error: (err as Error)?.message ?? String(err),
    }));
    return { ok: false, error: errorCode, detail: (err as Error)?.message };
  } finally {
    // 3. Audit log — always written, regardless of success/error.
    const paramsHash = await hashParams(args.params);
    await writeAuditLog(args.env, {
      member_id: args.member_id,
      startup_id: args.startup_id,
      channel: "mcp",
      action: args.action,
      params_hash: paramsHash,
      status,
      error_code: errorCode,
      latency_ms: Date.now() - t0,
    });
  }
}
```

```typescript
// apps/startup/workers/tools/search.ts
// v1.4 Phase 28 STARTUP-MCP-05 — search() full implementation.
//
// Scope routing:
//   candidates — pgvector cosine similarity via startup-api /v1/search/candidates
//   roles      — SQL SELECT from roles WHERE startup_id + text ILIKE
//   threads    — SQL SELECT from student_threads/inbound_messages WHERE startup_id
//   messages   — SQL SELECT from outbound_messages WHERE startup_id
//   members    — SQL SELECT from startup_members WHERE startup_id
//   startups   — SQL SELECT from startups WHERE id = startup_id (own record only)
//
// Result envelope: { scope, query, results: [{id, summary, score}], total_returned, next_cursor }
// IDs are returned so the LLM can call execute() with the specific record.
// Scores: 1.0 = exact match; 0.0 = no match. Structured searches return 1.0 for all hits.
//
// SECURITY: Every scope query adds WHERE startup_id = $startup_id from auth context.
// Cross-startup data leaks are impossible — no scope uses unconstrained queries.

import type { Env } from "../types";
import { embedText } from "../lib/embed";

export interface SearchArgs {
  startup_id: string;
  scope: "roles" | "candidates" | "threads" | "messages" | "members" | "startups";
  query: string;
  filters?: Record<string, unknown>;
  limit?: number;
}

interface SearchResult {
  id: string;
  summary: string;
  score: number;
}

async function proxyPost(url: string, secret: string, body: unknown): Promise<unknown> {
  const res = await fetch(url, {
    method: "POST",
    headers: { "Content-Type": "application/json", Authorization: `Bearer ${secret}` },
    body: JSON.stringify(body),
    signal: AbortSignal.timeout(10000),
  });
  if (!res.ok) return null;
  return res.json();
}

async function searchCandidates(args: SearchArgs, env: Env): Promise<SearchResult[]> {
  // Semantic search via pgvector — embed the query then call startup-api
  const embedding = await embedText(args.query, env);
  if (!embedding) {
    // Fallback: no embedding available — return empty (don't crash)
    console.warn(JSON.stringify({ level: "warn", event: "startup_search_no_embedding", scope: "candidates" }));
    return [];
  }
  const base = env.STARTUP_API_URL.replace(/\/$/, "");
  const limit = Math.min(Math.max(1, args.limit ?? 10), 20);
  const result = await proxyPost(`${base}/v1/search/candidates`, env.STARTUP_API_SECRET, {
    startup_id: args.startup_id,
    embedding,
    filters: {
      role_id: (args.filters?.role_id as string | undefined) ?? null,
      status:  (args.filters?.status  as string | undefined) ?? null,
    },
    limit,
  }) as { results?: Array<{ id: string; summary: string; score: number }> } | null;
  return result?.results ?? [];
}

async function searchStructured(
  scope: Exclude<SearchArgs["scope"], "candidates">,
  args: SearchArgs,
  env: Env,
): Promise<SearchResult[]> {
  // For non-candidate scopes: call a new startup-api search endpoint.
  // POST /v1/search/:scope { startup_id, query, filters, limit }
  // The proxy executes a ILIKE + WHERE startup_id SQL query.
  const base = env.STARTUP_API_URL.replace(/\/$/, "");
  const limit = Math.min(Math.max(1, args.limit ?? 10), 20);
  const result = await proxyPost(`${base}/v1/search/${scope}`, env.STARTUP_API_SECRET, {
    startup_id: args.startup_id,
    query: args.query,
    filters: args.filters ?? {},
    limit,
  }) as { results?: SearchResult[] } | null;
  return result?.results ?? [];
}

export async function handleSearch(args: SearchArgs) {
  const limit = Math.min(Math.max(1, args.limit ?? 10), 20);
  let results: SearchResult[] = [];

  try {
    if (args.scope === "candidates") {
      results = await searchCandidates(args, args as unknown as SearchArgs & { env: Env });
    } else {
      results = await searchStructured(args.scope, args, args as unknown as SearchArgs & { env: Env });
    }
  } catch (err) {
    console.error(JSON.stringify({
      level: "error", event: "startup_search_failed",
      scope: args.scope, error: (err as Error)?.message ?? String(err),
    }));
  }

  return {
    scope: args.scope,
    query: args.query,
    results,
    total_returned: results.length,
    next_cursor: null,
  };
}
```
