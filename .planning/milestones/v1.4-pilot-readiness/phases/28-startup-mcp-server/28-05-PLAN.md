```typescript
// apps/startup/workers/routes/api.ts
// v1.4 Phase 28 STARTUP-MARKETING-01
//
// POST /api/request-access — receives the "/startups" marketing CTA form.
// Body: { name, email, phone, what_hiring_for }
// Action: emails Ridhi at raj@internjobs.ai via CF Email Service (if EMAIL binding exists)
//         OR logs the lead if no email binding.
// Response: { ok: true } always (fail-safe — form submitter sees success even if email fails).

import { Hono } from "hono";
import { cors } from "hono/cors";
import type { Env } from "../types";

export const apiRouter = new Hono<{ Bindings: Env }>();

// CORS — marketing page is a different origin (Cloudflare Pages / internjobs.ai)
apiRouter.use("/request-access", cors({
  origin: ["https://internjobs.ai", "https://www.internjobs.ai"],
  allowMethods: ["POST", "OPTIONS"],
}));

apiRouter.post("/request-access", async (c) => {
  let body: { name?: string; email?: string; phone?: string; what_hiring_for?: string };
  try { body = await c.req.json(); }
  catch { return c.json({ error: "invalid_json" }, 400); }

  const { name, email, phone, what_hiring_for } = body;
  if (!name || !email) return c.json({ error: "name_and_email_required" }, 400);

  const subject = `startup access request — ${name} <${email}>`;
  const text = [
    `new startup access request from the /startups page.`,
    ``,
    `name: ${name}`,
    `email: ${email}`,
    `phone: ${phone ?? "(not provided)"}`,
    `what hiring for: ${what_hiring_for ?? "(not provided)"}`,
    ``,
    `to onboard: POST /admin/startups/new with company, founder_email, founder_phone`,
  ].join("\n");

  // Attempt CF Email Service send (EMAIL binding — same as Parrot uses)
  const emailBinding = (c.env as unknown as Record<string, unknown>).EMAIL;
  if (emailBinding && typeof (emailBinding as { send?: unknown }).send === "function") {
    try {
      await (emailBinding as { send: (msg: unknown) => Promise<void> }).send({
        from: { email: "noreply@internjobs.ai", name: "internjobs startup" },
        to: [{ email: "raj@internjobs.ai" }],
        subject,
        text,
      });
      console.log(JSON.stringify({ level: "info", event: "startup_access_request_emailed", from: email }));
    } catch (err) {
      console.warn(JSON.stringify({ level: "warn", event: "startup_access_request_email_failed", error: (err as Error)?.message ?? String(err) }));
    }
  } else {
    // No email binding — log the lead so Ridhi can see it in wrangler tail
    console.log(JSON.stringify({ level: "info", event: "startup_access_request_logged", name, email, phone, what_hiring_for }));
  }

  return c.json({ ok: true, message: "request received — ridhi will text you shortly." });
});
```
