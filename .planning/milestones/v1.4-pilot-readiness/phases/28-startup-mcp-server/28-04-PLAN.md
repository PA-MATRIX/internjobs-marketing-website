```typescript
// apps/startup/workers/routes/admin.ts
// v1.4 Phase 28 STARTUP-ADMIN-01 + STARTUP-ADMIN-02
//
// POST /admin/startups/new — Ridhi's concierge onboarding endpoint.
//
// Auth: Authorization: Bearer <STARTUP_MCP_ADMIN_SECRET> (pre-shared secret,
// separate from the per-startup install tokens). Ridhi-only access.
//
// Flow:
//   1. Validate admin secret (constant-time, crypto.subtle.timingSafeEqual)
//   2. Call startup-api POST /v1/startups → { startup_id, member_id, token }
//   3. Send SMS to founder_phone with install snippet (ctx.waitUntil — non-blocking)
//   4. Return { startup_id, member_id, token, install_snippet } in response body
//      (token visible in response body so Ridhi can copy it if SMS fails)
//
// Install snippet format (per research, locked decision #6):
//   claude mcp add --transport http internjobs https://mcp.internjobs.ai/mcp \
//     --header "Authorization: Bearer {TOKEN}"
//
// The response body also includes the Cursor .mcp.json format and a ChatGPT note.

import { Hono } from "hono";
import type { Env } from "../types";

// ── Admin secret verification ─────────────────────────────────────────────────

async function verifyAdminSecret(provided: string, env: Env): Promise<boolean> {
  const secret = env.STARTUP_MCP_ADMIN_SECRET;
  if (!secret || !provided) return false;
  const enc = new TextEncoder();
  const a = enc.encode(provided);
  const b = enc.encode(secret);
  if (a.byteLength !== b.byteLength) {
    // Constant time: compare a against itself to avoid length timing leak
    return !crypto.subtle.timingSafeEqual(a, a);
  }
  return crypto.subtle.timingSafeEqual(a, b);
}

// ── Install snippet builder ────────────────────────────────────────────────────

function buildInstallSnippet(token: string): string {
  return [
    `hi — you're set up on internjobs! here's how to connect:`,
    ``,
    `claude desktop / claude code:`,
    `claude mcp add --transport http internjobs https://mcp.internjobs.ai/mcp --header "Authorization: Bearer ${token}"`,
    ``,
    `cursor / cline (add to .mcp.json):`,
    `{"mcpServers":{"internjobs":{"type":"http","url":"https://mcp.internjobs.ai/mcp","headers":{"Authorization":"Bearer ${token}"}}}}`,
    ``,
    `chatgpt: connect via MCP connector — url: https://mcp.internjobs.ai/mcp, header: Authorization: Bearer ${token}`,
    ``,
    `start with: me() to confirm your identity, then discover_actions() to see what you can do.`,
    ``,
    `save this token — it won't be shown again. reply STOP to opt out.`,
  ].join("\n");
}

// ── SMS send helper ───────────────────────────────────────────────────────────

async function sendSmsTelnyxOrFallback(to: string, body: string, env: Env): Promise<void> {
  // Option A: Telnyx REST API (if TELNYX_API_KEY + TELNYX_FROM_NUMBER are bound)
  const telnyxKey    = (env as unknown as Record<string, string>).TELNYX_API_KEY;
  const telnyxFrom   = (env as unknown as Record<string, string>).TELNYX_FROM_NUMBER;

  if (telnyxKey && telnyxFrom) {
    try {
      const res = await fetch("https://api.telnyx.com/v2/messages", {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          Authorization: `Bearer ${telnyxKey}`,
        },
        body: JSON.stringify({
          from: telnyxFrom,
          to,
          text: body,
          messaging_profile_id: (env as unknown as Record<string, string>).TELNYX_MESSAGING_PROFILE_ID ?? undefined,
        }),
        signal: AbortSignal.timeout(10000),
      });
      if (res.ok) {
        console.log(JSON.stringify({ level: "info", event: "startup_admin_sms_sent_telnyx", to }));
        return;
      }
      const err = await res.text();
      console.warn(JSON.stringify({ level: "warn", event: "startup_admin_sms_telnyx_failed", status: res.status, error: err.slice(0, 200), to }));
    } catch (err) {
      console.warn(JSON.stringify({ level: "warn", event: "startup_admin_sms_telnyx_error", error: (err as Error)?.message ?? String(err), to }));
    }
  }

  // Option B: Student app /internal/sms if available (future)
  // Option C: Log token — Ridhi can manually copy from response body
  console.log(JSON.stringify({ level: "info", event: "startup_admin_sms_fallback_log", to, note: "SMS not sent — token is in response body" }));
}

// ── Hono router ───────────────────────────────────────────────────────────────

export const adminRouter = new Hono<{ Bindings: Env }>();

// POST /admin/startups/new
adminRouter.post("/startups/new", async (c) => {
  // 1. Auth
  const raw = c.req.header("Authorization")?.replace(/^Bearer\s+/i, "");
  if (!raw || !(await verifyAdminSecret(raw, c.env))) {
    return c.json({ error: "unauthorized" }, 401);
  }

  // 2. Parse body
  let body: { company?: string; founder_email?: string; founder_phone?: string };
  try { body = await c.req.json(); }
  catch { return c.json({ error: "invalid_json" }, 400); }

  const { company, founder_email, founder_phone } = body;
  if (!company || !founder_email || !founder_phone) {
    return c.json({ error: "company_founder_email_founder_phone_required" }, 400);
  }

  // 3. Create startup + member via startup-api proxy
  const base = c.env.STARTUP_API_URL.replace(/\/$/, "");
  let startup: { startup_id: string; member_id: string; token: string };
  try {
    const res = await fetch(`${base}/v1/startups`, {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        Authorization: `Bearer ${c.env.STARTUP_API_SECRET}`,
      },
      body: JSON.stringify({ company, founder_email, founder_phone }),
      signal: AbortSignal.timeout(10000),
    });

    if (res.status === 409) return c.json({ error: "startup_already_registered" }, 409);
    if (!res.ok) {
      const err = await res.text();
      console.error(JSON.stringify({ level: "error", event: "startup_admin_create_failed", status: res.status, error: err.slice(0, 200) }));
      return c.json({ error: "create_failed" }, 502);
    }

    startup = (await res.json()) as { startup_id: string; member_id: string; token: string };
  } catch (err) {
    console.error(JSON.stringify({ level: "error", event: "startup_admin_proxy_error", error: (err as Error)?.message ?? String(err) }));
    return c.json({ error: "proxy_unavailable" }, 503);
  }

  // 4. Build install snippet
  const installSnippet = buildInstallSnippet(startup.token);

  // 5. SMS send (non-blocking — fire-and-forget via waitUntil)
  c.executionCtx.waitUntil(
    sendSmsTelnyxOrFallback(founder_phone, installSnippet, c.env)
  );

  console.log(JSON.stringify({
    level: "info",
    event: "startup_admin_onboarded",
    startup_id: startup.startup_id,
    company,
  }));

  // 6. Return token in response body (Ridhi can manually send if SMS fails)
  return c.json({
    ok: true,
    startup_id: startup.startup_id,
    member_id: startup.member_id,
    // Token returned plaintext once — Ridhi copies it for manual follow-up if SMS fails
    token: startup.token,
    install_snippet: {
      claude_code: `claude mcp add --transport http internjobs https://mcp.internjobs.ai/mcp --header "Authorization: Bearer ${startup.token}"`,
      cursor_mcp_json: {
        mcpServers: {
          internjobs: {
            type: "http",
            url: "https://mcp.internjobs.ai/mcp",
            headers: { Authorization: `Bearer ${startup.token}` },
          },
        },
      },
      chatgpt: `Connect via MCP connector — url: https://mcp.internjobs.ai/mcp, header: Authorization: Bearer ${startup.token}`,
    },
    sms_sent_to: founder_phone,
    note: "Token shown here once. Not stored in plaintext. Save it.",
  });
});
```
