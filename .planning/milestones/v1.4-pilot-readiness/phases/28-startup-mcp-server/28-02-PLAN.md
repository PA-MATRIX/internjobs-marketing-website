// apps/startup/workers/lib/auth.ts
// v1.4 Phase 28 STARTUP-MCP-02 — Per-startup Bearer token validation.
//
// Auth flow:
//   1. Extract raw token from Authorization: Bearer <token> header
//   2. SHA-256 hash the raw token (crypto.subtle — CF Workers WebCrypto)
//   3. POST to internjobs-startup-api /v1/startups/token with the hash
//   4. Proxy returns { startup_id, member_id, startup_name } or 404
//   5. Constant-time double-check via crypto.subtle.timingSafeEqual (defense in depth)
//
// SECURITY NOTES:
//   - Token is ALWAYS in Authorization: Bearer header, never in URL path (leaks in logs)
//   - Rate limit by startup_id (stable across token rotation), not by token hash
//   - timingSafeEqual from crypto.subtle (Workers runtime) — NOT node:crypto

import type { Env, StartupContext } from "../types";

/** SHA-256 the raw token → hex string (64 chars). */
async function hashToken(rawToken: string): Promise<string> {
  const enc = new TextEncoder();
  const digest = await crypto.subtle.digest("SHA-256", enc.encode(rawToken));
  const bytes = new Uint8Array(digest);
  let hex = "";
  for (let i = 0; i < bytes.length; i++) hex += bytes[i].toString(16).padStart(2, "0");
  return hex;
}

/** Constant-time string equality using WebCrypto (CF Workers runtime). */
async function safeEqual(a: string, b: string): Promise<boolean> {
  const enc = new TextEncoder();
  const aBytes = enc.encode(a);
  const bBytes = enc.encode(b);
  if (aBytes.byteLength !== bBytes.byteLength) {
    // Still call timingSafeEqual to avoid timing correlation on length mismatch.
    return !crypto.subtle.timingSafeEqual(aBytes, aBytes);
  }
  return crypto.subtle.timingSafeEqual(aBytes, bBytes);
}

/**
 * Validates the incoming Authorization: Bearer token against the startup DB.
 * Returns StartupContext on success, null on failure.
 * Rate-limits by startup_id (stable across rotation) via CF Rate Limiting API if bound.
 */
export async function validateBearerToken(rawToken: string, env: Env): Promise<StartupContext | null> {
  if (!rawToken || !env.STARTUP_API_URL || !env.STARTUP_API_SECRET) return null;

  const tokenHash = await hashToken(rawToken);

  let row: StartupContext & { mcp_token_hash?: string } | null = null;
  try {
    const res = await fetch(`${env.STARTUP_API_URL.replace(/\/$/, "")}/v1/startups/token`, {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        Authorization: `Bearer ${env.STARTUP_API_SECRET}`,
      },
      body: JSON.stringify({ token_hash: tokenHash }),
      signal: AbortSignal.timeout(5000),
    });
    if (!res.ok) return null;
    row = (await res.json()) as StartupContext;
  } catch {
    return null;
  }

  if (!row?.startup_id || !row?.member_id) return null;
  return { startup_id: row.startup_id, member_id: row.member_id, startup_name: row.startup_name };
}

```typescript
// apps/startup/workers/server.ts
// v1.4 Phase 28 STARTUP-MCP-01..06 — Startup MCP server factory.
//
// SECURITY CRITICAL: createStartupMcpServer() MUST be called inside the
// fetch handler (per request), never at module level. A shared McpServer
// instance causes cross-client data leaks (SDK 1.26.0+ security note).
//
// Transport: Streamable HTTP (SSE deprecated March 2025).
// Tools: me, discover_actions, search, execute (4 tools, catalog fixed).

import { createMcpHandler, getMcpAuthContext } from "agents/mcp";
import { McpServer } from "@modelcontextprotocol/sdk/server/mcp.js";
import { z } from "zod";
import type { Env, StartupContext } from "./types";
import { handleMe } from "./tools/me";
import { handleDiscoverActions } from "./tools/discover-actions";
import { handleSearch } from "./tools/search";
import { handleExecute } from "./tools/execute";

/** Wrap a plain result as MCP tool output. */
export function mcpText(result: unknown) {
  return { content: [{ type: "text" as const, text: JSON.stringify(result, null, 2) }] };
}

/** Wrap an error as MCP tool output (isError signals tool-level failure, not protocol error). */
export function mcpError(message: string, code?: string) {
  return {
    content: [{ type: "text" as const, text: JSON.stringify({ error: message, code }) }],
    isError: true as const,
  };
}

/** Create a fresh McpServer per request. NEVER call this at module scope. */
export function createStartupMcpServer() {
  const server = new McpServer({
    name: "internjobs-startup",
    version: "1.0.0",
    instructions: "internjobs — startup hiring tools. Use these to post roles, search candidates, and reply to candidates. Start with me() to confirm your identity, then discover_actions() to see what you can do.",
  });

  // ── me ─────────────────────────────────────────────────────────────────────
  server.tool("me", "Returns your startup identity: name, member info, active role count, and recent activity summary.", {}, async () => {
    const ctx = getMcpAuthContext<StartupContext & { env: Env }>();
    if (!ctx) return mcpError("not_authenticated");
    return mcpText(await handleMe(ctx.props));
  });

  // ── discover_actions ───────────────────────────────────────────────────────
  server.tool("discover_actions", "Lists all available write actions with their JSON input schemas. Call this before execute() to see what actions are available and what parameters each requires.", {}, async () => {
    return mcpText(handleDiscoverActions());
  });

  // ── search ─────────────────────────────────────────────────────────────────
  server.tool(
    "search",
    "Semantic + structured search. scope must be one of: roles | candidates | threads | messages | members | startups. Returns {id, summary, score} list. Use IDs with execute() for actions.",
    {
      scope: z.enum(["roles", "candidates", "threads", "messages", "members", "startups"]).describe("What to search"),
      query: z.string().describe("Natural language search query"),
      filters: z.record(z.unknown()).optional().describe("Structured filters e.g. {role_id: 'uuid', status: 'active'}"),
      limit: z.number().int().min(1).max(20).default(10).describe("Max results (1–20, default 10)"),
    },
    async ({ scope, query, filters, limit }) => {
      const ctx = getMcpAuthContext<StartupContext & { env: Env }>();
      if (!ctx) return mcpError("not_authenticated");
      return mcpText(await handleSearch({ startup_id: ctx.props.startup_id, scope, query, filters, limit }));
    },
  );

  // ── execute ────────────────────────────────────────────────────────────────
  server.tool(
    "execute",
    "Execute a write action. action must be one of: post_role | reply_to_candidate | update_role | archive_role | mark_candidate. Call discover_actions() first to see parameter schemas.",
    {
      action: z.enum(["post_role", "reply_to_candidate", "update_role", "archive_role", "mark_candidate"]).describe("Action to execute"),
      params: z.record(z.unknown()).describe("Action-specific parameters (see discover_actions for schemas)"),
    },
    async ({ action, params }) => {
      const ctx = getMcpAuthContext<StartupContext & { env: Env }>();
      if (!ctx) return mcpError("not_authenticated");
      return mcpText(await handleExecute({ startup_id: ctx.props.startup_id, member_id: ctx.props.member_id, action, params, env: ctx.props.env }));
    },
  );

  return server;
}

/** Build the createMcpHandler for a validated startup context. New server per request. */
export function buildMcpHandler(env: Env, startupCtx: StartupContext) {
  return createMcpHandler(createStartupMcpServer(), {
    authContext: { props: { ...startupCtx, env } },
  });
}
```

// apps/startup/workers/app.ts
// v1.4 Phase 28 STARTUP-MCP-01..02 — Startup MCP Worker root.
//
// Routes:
//   /mcp        — MCP server (Streamable HTTP, per-startup Bearer auth)
//   /mcp/*      — MCP subpaths (some clients POST to /mcp/messages)
//   /admin/*    — Admin endpoint (Plan 28-04)
//   /api/*      — Marketing CTA receiver (Plan 28-05)
//   /.well-known/oauth-authorization-server — ChatGPT OAuth probe → 404 JSON
//
// Auth model: Authorization: Bearer <token> header on /mcp routes.
// The token is hashed and looked up via the startup-api proxy.
// Admin routes use a separate STARTUP_MCP_ADMIN_SECRET.

import { Hono } from "hono";
import { cors } from "hono/cors";
import { buildMcpHandler } from "./server";
import { validateBearerToken } from "./lib/auth";
import type { Env, StartupContext } from "./types";

const app = new Hono<{ Bindings: Env; Variables: { startupCtx: StartupContext } }>();

// CORS: MCP desktop clients are not browsers, but Cursor's webview may send preflight.
app.use("/mcp", cors({ origin: "*", allowMethods: ["GET", "POST", "OPTIONS"] }));
app.use("/mcp/*", cors({ origin: "*", allowMethods: ["GET", "POST", "OPTIONS"] }));

// Bearer auth middleware for /mcp routes
app.use("/mcp", async (c, next) => {
  const raw = c.req.header("Authorization")?.replace(/^Bearer\s+/i, "");
  if (!raw) return c.json({ error: "missing_bearer_token" }, 401);
  const ctx = await validateBearerToken(raw, c.env);
  if (!ctx) return c.json({ error: "invalid_token" }, 401);
  c.set("startupCtx", ctx);
  return next();
});
app.use("/mcp/*", async (c, next) => {
  // Streamable HTTP may POST to /mcp/messages or similar subpaths.

  if (!raw) return c.json({ error: "missing_bearer_token" }, 401);
  const ctx = await validateBearerToken(raw, c.env);
  if (!ctx) return c.json({ error: "invalid_token" }, 401);
  c.set("startupCtx", ctx);

});

// MCP handler — new server per request (SDK 1.26.0+ security requirement)
app.all("/mcp", async (c) => {
  const startupCtx = c.get("startupCtx");
  const handler = buildMcpHandler(c.env, startupCtx);
  return handler(c.req.raw, c.env, c.executionCtx as ExecutionContext);

  const handler = buildMcpHandler(c.env, startupCtx);

// ChatGPT OAuth probe — return 404 JSON so ChatGPT falls back to header auth.
app.get("/.well-known/oauth-authorization-server", (c) => {
