**Section 3 — Replace the connection singleton and getGraphClient().**

Replace the entire connection section with a fetch-based "proxy client" factory:

```typescript
// ─── HTTP proxy client ────────────────────────────────────────────────────────
//
// Phase 18 v1.3: The Parrot Worker cannot hold the falkordb npm client
// (crashes at module init: "e.BigInt is not a function"). All Cypher calls
// are now proxied via internjobs-graph-api (a Fly/Node app that holds the
// client). The Worker sends: POST /query { cypher, params } with a Bearer
// token → receives { data, stats } in return.
//
// This factory returns a "proxy graph object" that satisfies the
// `{ query: (cypher, opts) => Promise<{data, stats}> }` contract expected
// by every caller in this file. It is NOT a real FalkorDB client; it is a
// thin fetch wrapper.
//
// Fail-soft: if GRAPH_API_URL is unset or the proxy returns a non-2xx,
// every caller gets the same safe default (null / [] / "") it would get
// if the DB were down. No throws.

function makeProxyGraph(env: GraphEnv): {
  query: <T = unknown>(
    cypher: string,
    opts?: { params?: Record<string, unknown> },
  ) => Promise<{ data: T[]; stats: Record<string, unknown> } | null>;
} | null {
  if (!env.GRAPH_API_URL || !env.GRAPH_API_SECRET) return null;
  const url = env.GRAPH_API_URL.replace(/\/$/, "") + "/query";
  const secret = env.GRAPH_API_SECRET;

  return {
    async query<T>(
      cypher: string,
      opts?: { params?: Record<string, unknown> },
    ): Promise<{ data: T[]; stats: Record<string, unknown> } | null> {
      try {
        const res = await fetch(url, {
          method: "POST",
          headers: {
            "Content-Type": "application/json",
            Authorization: `Bearer ${secret}`,
          },
          body: JSON.stringify({ cypher, params: opts?.params ?? {} }),
          signal: AbortSignal.timeout(8000), // 8s budget — graph reads on dashboard path
        });
        if (!res.ok) {
          console.warn(
            JSON.stringify({
              level: "warn",
              message: "parrot_graph_proxy_error",
              status: res.status,
              cypher: cypher.slice(0, 120),
            }),
          );
          return null;
        }
        return (await res.json()) as { data: T[]; stats: Record<string, unknown> };
      } catch (err) {
        console.warn(
          JSON.stringify({
            level: "warn",
            message: "parrot_graph_proxy_fetch_failed",
            error: (err as Error)?.message ?? String(err),
            cypher: cypher.slice(0, 120),
          }),
        );
        return null;
      }
    },
  };
}

// Helper used by all exported functions. Returns null if not configured.
function getProxyGraph(env: GraphEnv) {
  return makeProxyGraph(env);
}
```

**Section 6 — Update `pingParrotGraph()`.**

Old version called `getGraphClient(env)` then `graph.query("RETURN 1")`.
New version calls GET /health on the proxy (which itself probes FalkorDB):

```typescript
/**
 * Proxy health probe. Returns true iff the internjobs-graph-api proxy is
 * reachable AND it reports ok:true (meaning FalkorDB is also reachable).
 * Used by /healthz as `graph_proxy_reachable` (distinct from `graph_ready`
 * which reflects the student app's direct FalkorDB ping).
 */
export async function pingParrotGraph(env: GraphEnv): Promise<boolean> {
  if (!env.GRAPH_API_URL) return false;
  try {
    const res = await fetch(
      env.GRAPH_API_URL.replace(/\/$/, "") + "/health",
      { signal: AbortSignal.timeout(3000) },
    );
    if (!res.ok) return false;
    const body = (await res.json()) as { ok?: boolean };
    return body?.ok === true;
  } catch {
    return false;
  }
}
```
